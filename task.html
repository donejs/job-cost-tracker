export const Task = can.Map.extend({
  define: {
    name: { type: 'string' },
    hours: { type: 'number' },
    cubicYards: { type: 'number' },
    tons: { type: 'number' },
    notes: { type: 'string' },
    completed: { type: 'date' },
    Job: { Type: Job },
    Lot: { Type: Lot }
  }
});

Task.List = can.List.extend({
  Map: Task
}, {
  totals: function(){
    var totals = {
      cubicYards: 0,
      tons: 0,
      hours: 0
    };

    this.each(function(task){
      totals.cubicYards += task.attr('cubicYards');
      totals.tons += task.attr('tons');
      totals.hours += task.attr('hours');
    });

    return totals;
  }
});
