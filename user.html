class TokenStore {
  constructor({ key="id_token" } = {}) {
    this.key = key;
  }
  fetchToken() {
    return Promise.resolve(localStorage.getItem(this.key));
  }
  setToken(token) {
    return Promise.resolve(localStorage.setItem(this.key, token));
  }
  deleteToken() {
    return Promise.resolve(localStorage.removeItem(this.key));
  }
}
